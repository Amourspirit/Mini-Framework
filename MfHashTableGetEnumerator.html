GetEnumerator()

Returns an enumerator.

Remarks

Returns an enumerator that iterates through a collection of key/value pairs.

Example

msd := new MfDictionary()
loop, 10
{
       msd.Add("MyKey" . A_Index, "MyValue-" . A_Index)
}
MyEnum := msd.GetEnumerator()

mfs := new MfString()
MaxIndex := msd.Count -1
i := 0
While MyEnum.Next(key, value)
{
   mfs.Append(MfString.Format("Key: {0}, Value: '{1}'", key, value))
   if (i < MaxIndex)
   {
       mfs.AppendLine()
   }
   i++
}

MsgBox % mfs.Value

; alternatively we can do this
mfs := new MfString()
MaxIndex := msd.Count -1
i := 0
for key, value in msd
{
   mfs.Append(MfString.Format("Key: {0}, Value: '{1}'", Key, value))
   if (i < MaxIndex)
   {
       mfs.AppendLine()
   }
   i++
}

MsgBox % mfs.Value

; mfs.Value contains
/*
Key: MyKey1, Value: 'MyValue-1'
Key: MyKey10, Value: 'MyValue-10'
Key: MyKey2, Value: 'MyValue-2'
Key: MyKey3, Value: 'MyValue-3'
Key: MyKey4, Value: 'MyValue-4'
Key: MyKey5, Value: 'MyValue-5'
Key: MyKey6, Value: 'MyValue-6'
Key: MyKey7, Value: 'MyValue-7'
Key: MyKey8, Value: 'MyValue-8'
Key: MyKey9, Value: 'MyValue-9'
*/
; Note that MfDictionary enumerator does not guarantee sort order
