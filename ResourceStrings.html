Related

Example

Class MyClass extends MfObject
{
   __New()
   {
   if (this.__Class != "MyClass")
   {
       throw new MfNotSupportedException(MfEnvironment.Instance.GetResourceString("NotSupportedException_Sealed_Class"
               , this.__Class))
   }
   Base.__New()
   }
}
