Example

mfQ := new MfQueue()
i := 0
iEnd := 5
While, (i < iEnd)
{
   mss := new MfString()
   mss.Value := MfString.Format("String Value: '{0}'", i)
   mfQ.Enqueue(mss)
   i++
}
mfsOutput := new MfString()
While mfQ.Count > 0
{
   mfsOutput.AppendLine(mfQ.Dequeue())
}
MsgBox, 64, Text, % mfsOutput.Value
/*
String Value: '0'
String Value: '1'
String Value: '2'
String Value: '3'
String Value: '4'

*/
